//código para que al presionarse el botón, se encienda el led, y al presionarse otra vez, 
//se apague

int PULSADOR=2;
int LED=3;
int ESTADO = LOW; 

void setup(){
  pinMode(PULSADOR, INPUT);     //PIN 2 COMO ENTRada
  pinMode(LED, OUTPUT);    //PIN 3 COMO SALIDA
}

void loop(){
  while (digitalRead(PULSADOR)==LOW){   //se espera q se presione el pulsador
  }
  ESTADO= digitalRead(LED);   //se lee el estado del led
  digitalWrite (LED, !ESTADO);  //se escribe el valor contrario con !
  while(digitalRead(PULSADOR)==HIGH){  //antirrebote 

  }
}
